# 2、请写出下面的结果，为什么

for (var i = 0; i < 5; i++) {
  setTimeout(function () {
    console.log(i)
  }, 1000 * 1)

# 解析：

  • var 声明的 i 是全局作用域，循环结束后 i=5
  • setTimeout 回调共享最终的 i，所有输出均为 5

# 3、请写出下面的输出结果，最好说明一下原因

    // 修正语法错误
    console.log(this.x)
  }

var f1 = new Fn()
console.log(f1.num) // undefined（局部变量无法通过实例访问）
console.log(num) // 报错：num未定义
